✨ Animações & Interações
Sistema de Animações
javascript
// 1. Scroll-triggered animations
const observeElements = () => {
  const elements = document.querySelectorAll('[data-animate]');
  const observer = new IntersectionObserver((entries) => {
    entries.forEach(entry => {
      if (entry.isIntersecting) {
        const delay = parseInt(entry.target.dataset.delay || '0');
        setTimeout(() => {
          entry.target.classList.add('opacity-100', 'translate-y-0', 'blur-0');
          entry.target.classList.remove('opacity-0');
        }, delay);
      }
    });
  }, { threshold: 0.2, rootMargin: '0px 0px -10% 0px' });
  
  elements.forEach(el => observer.observe(el));
};

🔧 Scripts & Funcionalidades
Funcionalidades Obrigatórias
javascript
// 1. Inicialização dos ícones Lucide
lucide.createIcons();

// 2. Smooth scroll para seções
const setupSmoothScroll = () => {
  document.querySelectorAll('[data-jump]').forEach(btn => {
    btn.addEventListener('click', () => {
      const sectionId = btn.dataset.jump;
      const section = document.querySelector(`[data-section="${sectionId}"]`);
      section?.scrollIntoView({ behavior: 'smooth', block: 'start' });
    });
  });
};

// 3. Progress sidebar ativo
const updateProgressIndicator = () => {
  const sections = document.querySelectorAll('[data-section]');
  const buttons = document.querySelectorAll('[data-jump]');
  
  const observer = new IntersectionObserver((entries) => {
    entries.forEach(entry => {
      if (entry.isIntersecting) {
        const sectionId = entry.target.dataset.section;
        buttons.forEach(btn => {
          btn.classList.toggle('is-active', btn.dataset.jump === sectionId);
        });
      }
    });
  }, { threshold: 0.6 });
  
  sections.forEach(section => observer.observe(section));
};

// 4. CTAs funcionais
const setupCTAs = () => {
  const ecosystemsSection = document.querySelector('[data-section="3"]');
  
  ['mainCTA', 'headerCTA', 'finalCTA'].forEach(id => {
    document.getElementById(id)?.addEventListener('click', () => {
      ecosystemsSection?.scrollIntoView({ behavior: 'smooth' });
    });
  });
};
